struct Node {
    int data;
    Node* next;
};

//     
Node* addNode(Node* head, int value) {
    Node* newNode = new Node{value, head};
    return newNode;
}

//  
void deleteList(Node* head) {
    while (head != nullptr) {
        Node* temp = head;
        head = head->next;
        delete temp;
    }
}

//  
void printList(const Node* head) {
    while (head != nullptr) {
        std::cout << head->data << " ";
        head = head->next;
    }
    std::cout << std::endl;
}

int main() {
    Node* head = nullptr;
    head = addNode(head, 3);
    head = addNode(head, 2);
    head = addNode(head, 1);

    printList(head);
    deleteList(head);
    return 0;
}
